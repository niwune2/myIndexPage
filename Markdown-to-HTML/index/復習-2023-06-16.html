アクティブラーニング: 発射のカウントダウン

for-loop - カウントダウン

let output = document.querySelector('.output');
output.innerHTML = '';

for(let i =10 ; i >= 0 ; i--){ //1
    const para = document.createElement('p');//2
    if (i === 10){ //3
        para.textContent = `発射まで ${i}`;//4
    }else if (i === 0){//5
        para.textContent = '発射';//6
    }else{//7
        para.textContent = i;//7
    }
    output.appendChild(para);//8
}
  1. //1: for(let i =10 ; i >= 0 ; i--)
    1. let i = 10 : カウントダウンするため、10から開始。
    2. i >= 0 : iが0以上の数である限り、ループを繰り返すという意味。
    3. i-- : ループを一周するたびに、iに1を減算する。
  2. //2: <p>要素を作るcreateElementを変数paraに格納
  3. //3: 条件式の始まり
    1. //3: iが10であるとき、専用のテキストを表示する(//4)
    2. //5: iが0であるとき、専用のテキストを表示する(//6)
    3. //7: iが10または0ではないとき、そのまま数値iを表示する
  4. //8: if ステートメント の条件式から出力(?)された数値がappendChild()メソッドでpara変数に追加される。
    1. appendChild()は、親ノードの子ノードリストの末尾に新たにノードを追加する。

do...while -　カウントダウン

let i = 10;
do {
    const para =document.createElement('p');
    if(i===10){
        para.textContent = `発射まで・・・${i}秒`;
    }else if(i===0){
        para.textContent = `発射〜`;
    }else{
        para.textContent = `${i}秒`;
    }
    output.appendChild(para);
    i--;
}while(i >= 0);

for (const person of people) {  //1
  if (person === 'Phil' || person === 'Lola') { //2
    refused.textContent += `${person}, `;   //3
  } else {  //4
    admitted.textContent += `${person}, `;  //5
  }
}
refused.textContent = refused.textContent.slice(0,refused.textContent.length-2) + '.';  //6
admitted.textContent = admitted.textContent.slice(0,admitted.textContent.length-2) + '.';   //7

  3. //3: resused.textContentでページ上のRefuseに加算代入(+=)する
    1. 変数personを使って該当する名前をリストに追加する
    2. テンプレートリテラルを用いて代入することで、コンマ(,)も合わせて記入できる
  4. //4: Refuse以外の名前はすべて下の処理が適用される
  5. //6,//7: リストの最後尾にピリオド(.)を追加するための2行

模範解答-3

let i = 500;    //1
const para = document.createElement('p');   //2
function isPrime(num) {     //3
    for(let i = 2; i < num; i++) {      //4
    if(num % i === 0) {     //5
        return false;       //6
    }
    }
    return true;    //7
}
do {    //8
  if (isPrime(i)) {     //9
    para.textContent += `${i} `;    //10
  }
  i--;  //11
} while (i > 1);    //12
const section = document.querySelector('section');  //13
section.appendChild(para);  //14
  1. //1: イテレーターiは500で指定
  2. //2: <p>要素の作成を変数paraに格納
  3. //3: isPrime(num)関数.素数を判定するための関数
    1. //4: let i = 2は、1が素数として判定されないため2から始めている
    2. //4: iがnumより小さい間はループを繰り返す
    3. //5: 剰余演算子%は、左項を右項で割った余りを求める
      1. 素数は、「1とその数以外で割り切れない数」なので、割って0になる、つまり=== 0になる数値は素数ではないと判定する
      2. //6: その値はfalseとしてreturnされる
    4. //7: falseとなった数値以外はすべて素数と判定されるのでtrueとなる
  4. //8: do...whileループ
  5. //9:isPrime()関数を条件式にしたifステートメント
    1. //1のiの数値のうち、isPrime関数が素数判定(true)した数値を条件として
    2. //10: テンプレートリテラルでテキストとしてその数値を表示する
  6. //11: 500から数えるためi--
  7. //12: iが1より大きい数までループを繰り返す
  8. //13: HTML内で<section>をquerySelectorで探す
    1. //14:その<section>に//2のparaを収める
    2. こうなっているはず<section class="preview"><p>(//10の数値)</p></section>
